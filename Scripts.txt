
-- Create tickets table
CREATE TABLE TICKETS (
    ID INT PRIMARY KEY IDENTITY(1,1),
    IS_RESERVED BIT DEFAULT 0,
    RESERVED_BY CHAR(50) NULL,
    EMPLOYEE_ID INT NULL,
    IS_WINNER BIT DEFAULT 0,
    CONTROL_TIMESTAMP DATETIME NULL,
);

-- Insert 100 tickets
DECLARE @i INT = 1;

WHILE @i <= 100
BEGIN
    INSERT INTO Tickets (IS_RESERVED, RESERVED_BY, IS_WINNER, CONTROL_TIMESTAMP, EMPLOYEE_ID)
    VALUES (0, NULL, 0, NULL, NULL);

    SET @i = @i + 1;
END
